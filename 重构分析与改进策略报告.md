### 2.1 决策权重分配机制重构（最高优先级）

#### 策略 2.1.1：实现决策权重计算引擎

**目标**: 完全重构决策机制，移除 IDE 仅有执行权的限制

**数学模型**:
$$W_{total} = W_{user} + W_{swarm} = 1$$
$$W_{user} = \frac{1}{3}$$
$$W_{swarm} = \frac{2}{3}$$
$$w_i = \frac{W_{swarm}}{N} = \frac{2}{3N} \quad (N \ge 5)$$

**实现步骤**:

1. **创建决策权重管理器** (`src/decision/weightManager.js`)
```javascript
export class WeightManager {
  constructor() {
    this.userWeight = 1/3;
    this.swarmWeight = 2/3;
    this.minParticipants = 5;
    this.participants = new Map(); // IDE_ID -> { weight, score, history }
  }

  // 计算单个 IDE 的决策权重
  calculateIndividualWeight(totalParticipants) {
    if (totalParticipants < this.minParticipants) {
      throw new Error('参与者数量不足，触发熔断');
    }
    return this.swarmWeight / totalParticipants;
  }

  // 注册协作主体
  registerParticipant(ideId, capabilities = []) {
    const weight = this.calculateIndividualWeight(this.participants.size + 1);
    this.participants.set(ideId, {
      weight,
      score: 0,
      history: [],
      capabilities
    });
    return weight;
  }

  // 更新参与者评分
  updateParticipantScore(ideId, delta) {
    const participant = this.participants.get(ideId);
    if (participant) {
      participant.score += delta;
      // 动态调整权重（可选）
      this.recalculateWeights();
    }
  }

  // 重新计算所有权重（基于评分）
  recalculateWeights() {
    const scores = Array.from(this.participants.values()).map(p => p.score);
    const totalScore = scores.reduce((sum, s) => sum + Math.max(0, s), 0);

    if (totalScore > 0) {
      this.participants.forEach((participant, ideId) => {
        participant.weight = (Math.max(0, participant.score) / totalScore) * this.swarmWeight;
      });
    }
  }
}
```

2. **创建投票仲裁器** (`src/decision/voter.js`)
```javascript
export class Voter {
  constructor(weightManager) {
    this.weightManager = weightManager;
    this.proposals = new Map();
  }

  // 创建提案
  createProposal(proposalId, description, options = []) {
    this.proposals.set(proposalId, {
      id: proposalId,
      description,
      options,
      votes: new Map(), // option -> { ideId, weight }
      status: 'PENDING',
      createdAt: Date.now()
    });
    return proposalId;
  }

  // 投票
  vote(proposalId, ideId, option) {
    const proposal = this.proposals.get(proposalId);
    if (!proposal) throw new Error('提案不存在');
    if (proposal.status !== 'PENDING') throw new Error('提案已关闭');

    const participant = this.weightManager.participants.get(ideId);
    if (!participant) throw new Error('参与者未注册');

    const votes = proposal.votes.get(option) || [];
    votes.push({ ideId, weight: participant.weight });
    proposal.votes.set(option, votes);

    // 检查是否达到法定人数
    const totalVotes = Array.from(proposal.votes.values())
      .flat().length;

    if (totalVotes >= this.weightManager.minParticipants) {
      return this.resolveProposal(proposalId);
    }

    return { status: 'CONTINUE' };
  }

  // 解决提案（计算加权得分）
  resolveProposal(proposalId) {
    const proposal = this.proposals.get(proposalId);
    proposal.status = 'RESOLVED';

    const results = [];
    let maxScore = -1;
    let winner = null;

    for (const [option, votes] of proposal.votes) {
      const totalWeight = votes.reduce((sum, v) => sum + v.weight, 0);
      results.push({ option, totalWeight, votes });

      if (totalWeight > maxScore) {
        maxScore = totalWeight;
        winner = option;
      }
    }

    // 检查是否超过用户权重
    if (maxScore < this.weightManager.userWeight) {
      proposal.winner = 'REJECTED'; // 未达到用户权重门槛
    } else {
      proposal.winner = winner;
    }

    proposal.results = results;
    proposal.resolvedAt = Date.now();

    return proposal;
  }

  // 获取提案结果
  getProposal(proposalId) {
    return this.proposals.get(proposalId);
  }
}
```

#### 策略 2.1.2：实现法定人数熔断机制

**目标**: 当参与投票的协作主体少于 5 个时，自动触发熔断

**实现步骤**:

1. **创建熔断器** (`src/decision/circuitBreaker.js`)
```javascript
export class CircuitBreaker {
  constructor(minParticipants = 5) {
    this.minParticipants = minParticipants;
    this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
    this.failures = 0;
    this.lastFailureTime = 0;
    this.recoveryTimeout = 60000; // 1 分钟后尝试恢复
  }

  // 检查是否可以执行投票
  canExecute(participantCount) {
    if (participantCount < this.minParticipants) {
      this.failures++;
      this.lastFailureTime = Date.now();
      this.state = 'OPEN';
      return false;
    }

    if (this.state === 'OPEN') {
      const now = Date.now();
      if (now - this.lastFailureTime > this.recoveryTimeout) {
        this.state = 'HALF_OPEN';
        return true;
      }
      return false;
    }

    return true;
  }

  // 记录成功
  recordSuccess() {
    this.failures = 0;
    this.state = 'CLOSED';
  }

  // 获取状态
  getState() {
    return {
      state: this.state,
      failures: this.failures,
      lastFailureTime: this.lastFailureTime
    };
  }

  // 手动重置
  reset() {
    this.state = 'CLOSED';
    this.failures = 0;
    this.lastFailureTime = 0;
  }
}
```

```javascript
export class PerformanceMonitor {
  constructor() {
    this.metrics = new Map(); // providerName -> { avgResponseTime, errorRate, totalCalls, successCalls }
  }

  recordCall(providerName, responseTime, success) {
    const metrics = this.metrics.get(providerName) || {
      avgResponseTime: 0,
      errorRate: 0,
      totalCalls: 0,
      successCalls: 0
    };

    metrics.totalCalls++;
    if (success) {
      metrics.successCalls++;
    }

    // 计算平均响应时间
    metrics.avgResponseTime =
      (metrics.avgResponseTime * (metrics.totalCalls - 1) + responseTime) /
      metrics.totalCalls;

    // 计算错误率
    metrics.errorRate = 1 - (metrics.successCalls / metrics.totalCalls);

    this.metrics.set(providerName, metrics);
  }

  getMetrics(providerName) {
    return this.metrics.get(providerName);
  }

  getRanking(level) {
    const providers = this.metrics.keys();
    const ranking = [];

    for (const provider of providers) {
      const metrics = this.metrics.get(provider);
      ranking.push({
        provider,
        ...metrics,
        score: this.calculateScore(metrics)
      });
    }

    // 按分数降序排序
    ranking.sort((a, b) => b.score - a.score);
    return ranking;
  }

  calculateScore(metrics) {
    // 分数 = 1 / (响应时间 * 1000 + 错误率 * 10000)
    return 1 / (metrics.avgResponseTime * 1000 + metrics.errorRate * 10000);
  }
}
```

```javascript
export class ToolCache {
  constructor() {

      read: new LRUCache({ max: 100, ttl: 5 * 60 * 1000 }), // 5 分钟
      fetch: new LRUCache({ max: 100, ttl: 10 * 60 * 1000 }), // 10 分钟

      status: new LRUCache({ max: 50, ttl: 30 * 1000 }), // 30 秒

    if (!cache) return null;

    const value = cache.get(key);
    if (value !== undefined) {
      this.stats.hits++;
      log(`[Cache] HIT: ${toolName}/${key}`, 'DEBUG');
      return value;
    }

    this.stats.misses++;
    log(`[Cache] MISS: ${toolName}/${key}`, 'DEBUG');
    return null;
  }

  set(toolName, key, value) {
    const cache = this.caches[toolName];
    if (!cache) return;

    cache.set(key, value);
    log(`[Cache] SET: ${toolName}/${key}`, 'DEBUG');
  }

  getStats() {
    const total = this.stats.hits + this.stats.misses;
    const hitRate = total > 0 ? (this.stats.hits / total * 100).toFixed(2) : 0;

    return {
      hits: this.stats.hits,
      misses: this.stats.misses,
      hitRate: `${hitRate}%`
    };
  }

  clear() {
    Object.values(this.caches).forEach(cache => cache.clear());
    this.stats = { hits: 0, misses: 0 };
    log('[Cache] Cleared all caches', 'INFO');
  }
}
```

```javascript
import { log } from './logger.js';

export class MemoryPredictor {
  constructor() {
    this.history = [];
    this.maxHistorySize = 100;
    this.thresholds = {
      low: 100 * 1024 * 1024,    // 100 MB
      medium: 200 * 1024 * 1024, // 200 MB
      high: 400 * 1024 * 1024    // 400 MB
    };
  }

  record(memoryUsage) {
    this.history.push({
      timestamp: Date.now(),
      heapUsed: memoryUsage.heapUsed,
      external: memoryUsage.external
    });

    if (this.history.length > this.maxHistorySize) {
      this.history.shift();
    }
  }

  predictNext() {
    if (this.history.length < 10) {
      return null;
    }

    // 简单的线性回归预测
    const recent = this.history.slice(-20);
    const avg = recent.reduce((sum, h) => sum + h.heapUsed, 0) / recent.length;

    // 计算趋势
    const trend = (recent[recent.length - 1].heapUsed - recent[0].heapUsed) / recent.length;

    // 预测下一个值
    const predicted = avg + trend * 5; // 预测 5 次调用后的值

    return {
      current: avg,
      predicted,
      trend,
      shouldOptimize: predicted > this.thresholds.medium
    };
  }

  shouldTriggerGC() {
    const prediction = this.predictNext();
    if (!prediction) return false;

    return prediction.shouldOptimize;
  }
}

export class MemoryManager {
  constructor() {
    this.predictor = new MemoryPredictor();
  }

  checkAndOptimize() {
    const usage = process.memoryUsage();
    this.predictor.record(usage);

    if (global.gc && this.predictor.shouldTriggerGC()) {
      const prediction = this.predictor.predictNext();
      log(`[Memory] 预测内存将达到 ${(prediction.predicted / 1024 / 1024).toFixed(2)} MB，触发 GC`, 'INFO');
      global.gc();
      return true;
    }

    return false;
  }

  getReport() {
    const usage = process.memoryUsage();
    const prediction = this.predictor.predictNext();

    return {
      heapUsed: `${(usage.heapUsed / 1024 / 1024).toFixed(2)} MB`,
      external: `${(usage.external / 1024 / 1024).toFixed(2)} MB`,
      rss: `${(usage.rss / 1024 / 1024).toFixed(2)} MB`,
      prediction: prediction ? `${(prediction.predicted / 1024 / 1024).toFixed(2)} MB` : 'N/A'
    };
  }
}
```

```javascript
export class QualityEvaluator {
  constructor() {
    this.metrics = new Map();
  }

  evaluate(ideId, changes, executionResult) {
    const metrics = {
      ideId,
      timestamp: Date.now(),
      scores: {
        codeQuality: this.evaluateCodeQuality(changes),
        executionSuccess: this.evaluateExecution(executionResult),
        testCoverage: this.evaluateTestCoverage(changes),
        documentation: this.evaluateDocumentation(changes)
      },
      totalScore: 0
    };

    // 计算总分
    const weights = {
      codeQuality: 0.4,
      executionSuccess: 0.3,
      testCoverage: 0.2,
      documentation: 0.1
    };

    metrics.totalScore = Object.entries(weights).reduce((sum, [key, weight]) => {
      return sum + metrics.scores[key] * weight;
    }, 0);

    return metrics;
  }

  evaluateCodeQuality(changes) {
    // 检查代码规范、复杂度等
    let score = 100;

    // 减分项：长函数
    if (changes.some(c => c.lines > 50)) score -= 10;

    // 减分项：深层嵌套
    if (changes.some(c => c.nestingDepth > 4)) score -= 10;

    // 减分项：魔法数字
    if (changes.some(c => c.magicNumbers > 5)) score -= 5;

    return Math.max(0, score);
  }

  evaluateExecution(result) {
    if (result.success) return 100;
    if (result.error && result.error.isRetryable) return 50;
    return 0;
  }

  evaluateTestCoverage(changes) {
    // 检查是否有对应的测试
    const hasTests = changes.some(c => c.path.includes('test'));
    return hasTests ? 100 : 0;
  }

  evaluateDocumentation(changes) {
    // 检查是否有注释和文档
    const hasComments = changes.some(c => c.hasComments);
    const hasDocs = changes.some(c => c.hasDocs);
    return (hasComments ? 50 : 0) + (hasDocs ? 50 : 0);
  }

  getAverageScore(ideId) {
    const history = this.metrics.get(ideId) || [];
    if (history.length === 0) return null;

    const total = history.reduce((sum, m) => sum + m.totalScore, 0);
    return (total / history.length).toFixed(2);
  }
}
```

```javascript
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export class ConflictDetector {
  constructor() {
    this.lockFile = '.git/IDE_LOCK';
  }

  async acquireLock(ideId) {
    try {
      // 检查是否已有锁
      const lockData = await this.readLock();
      if (lockData && lockData.ideId !== ideId) {
        const elapsed = Date.now() - lockData.timestamp;
        if (elapsed < 30000) { // 30 秒内
          throw new Error(
            `冲突检测: IDE ${lockData.ideId} 正在工作，请等待 ${(elapsed / 1000).toFixed(1)} 秒后重试`
          );
        }
      }

      // 获取锁
      await this.writeLock({ ideId, timestamp: Date.now() });
      return true;
    } catch (error) {
      throw new Error(`获取锁失败: ${error.message}`);
    }
  }

  async releaseLock(ideId) {
    const lockData = await this.readLock();
    if (lockData && lockData.ideId === ideId) {
      await this.deleteLock();
    }
  }

  async readLock() {
    try {
      const { stdout } = await execAsync(`type ${this.lockFile}`);
      return JSON.parse(stdout.trim());
    } catch (error) {
      return null;
    }
  }

  async writeLock(data) {
    await execAsync(`echo ${JSON.stringify(data)} > ${this.lockFile}`);
  }

  async deleteLock() {
    await execAsync(`del ${this.lockFile}`);
  }

  async checkConflicts() {
    try {
      const { stdout } = await execAsync('git status --porcelain');
      if (stdout.trim().length > 0) {
        return true; // 有未提交的变更
      }
      return false;
    } catch (error) {
      return false;
    }
  }
}
```

## 第七章：附录

### 7.1 决策权重数学模型

#### 基础公式

$$W_{total} = W_{user} + W_{swarm} = 1$$
$$W_{user} = \frac{1}{3}$$
$$W_{swarm} = \frac{2}{3}$$

#### 个人权重计算

$$w_i = \frac{W_{swarm}}{N} = \frac{2}{3N}$$

其中：
- $N$: 参与投票的协作主体数量
- $w_i$: 第 $i$ 个协作主体的权重

#### 动态权重调整

$$w_i^{new} = w_i^{base} \times (1 + \alpha \times \frac{s_i - \bar{s}}{\sigma_s})$$

其中：
- $w_i^{new}$: 调整后的权重
- $w_i^{base}$: 基础权重
- $s_i$: 协作主体 $i$ 的历史评分
- $\bar{s}$: 平均评分
- $\sigma_s$: 评分标准差
- $\alpha$: 调整系数（建议 0.1-0.3）

```
    ↓ (否)

    ↓ (是)
分配任务给所有协作主体
    ↓
IDE 1 执行并投票
    ↓
IDE 2 执行并投票
    ↓
...
    ↓
IDE N 执行并投票
    ↓
检查是否达到法定人数
    ↓ (否)
继续等待
    ↓ (是)
计算加权得分
    ↓
检查是否超过用户权重 (1/3)
    ↓ (否)
拒绝执行，返回给用户
    ↓ (是)
执行胜出方案
    ↓
更新协作主体评分
    ↓
返回结果
```
